int nilpotentmi (int boyut,int dizi[boyut][boyut] )
{
    int matrisinkaresi[boyut][boyut];
        int topla=0;
        for(int i=0 ; i<boyut ;i++)
        {
            for(int j=0 ; j<boyut ; j++)
            {
                topla=0;
                for(int k=0 ; k<boyut ; k++)
                {
                    topla += dizi[i][k]*dizi[k][j];
                }
                matrisinkaresi[i][j] = topla;
                //printf("%d   ",matrisinkaresi[i][j]);

            }
           //printf("\n");
        }
        int kontrol=1;
        for(int i=0 ; i<boyut ;i++)
        {
            for(int j=0 ; j<boyut ; j++)
            {
                if(matrisinkaresi[i][j]!=0)
                kontrol=0;
            }
        }


        return kontrol;
}

}
